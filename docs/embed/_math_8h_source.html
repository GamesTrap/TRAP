3885 TRAP::Math::tQuat<T> TRAP::Math::Pow(const tQuat<T>& x, T y)
3886 {

3888  //Needed to prevent a division by 0 error later on
3889  if (y > -Epsilon<T>() && y < Epsilon<T>())
3890  return tQuat<T>(1, 0, 0, 0);
3891 
3892  //To deal with non-unit quaternions
3893  T magnitude = Sqrt(x.x * x.x + x.y * x.y + x.z * x.z + x.w * x.w);
3894 
3895  T angle;
3896  if(Abs(x.w / magnitude) > CosOneOverTwo<T>())
3897  {
3898  //Scalar component is close to 1; using it to recover angle would lose precision

3900
