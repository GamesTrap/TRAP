3691 template <typename T>
3692 TRAP::Math::tQuaternion<T> TRAP::Math::Pow(const tQuaternion<T>& x, T y)
3693 {
3694  //Raising to the power of 0 should yield 1
3695  //Needed to prevent a division by 0 error later on
3696  if (y > -Epsilon<T>() && y < Epsilon<T>())
3697  return tQuaternion<T>(1, 0, 0, 0);
3698 
3699  //To deal with non-unit quaternions
3700  T magnitude = Sqrt(x.x * x.x + x.y * x.y + x.z * x.z + x.w * x.w);
3701 
3702  T angle;
3703  if(Abs(x.w / magnitude) > CosOneOverTwo<T>())
3704  {
3705  //Scalar component is close to 1; using it to recover angle would lose precision
3706  //Instead, we use the non-scalar components since Sin() is accurate around 0
3707 
3708  //Prevent a division by 0 error later on
3709  T vectorMagnitude = x.x * x.x + x.y * x.y + x.z * x.z;
3710  //if(Abs(vectorMagnitude - static_cast<T>(0)) < Epsilon<T>())
3711  //Despite the compiler might say, we actually want to compare vectorMagnitude to 0.
3712  if(vectorMagnitude < std::numeric_limits<T>::min())
3713  {
3714  //Equivalent to raising a real number to a power
3715  return tQuaternion<T>(Pow(x.w, y), 0, 0, 0);
3716  }
3717 
3718  angle = ASin(Sqrt(vectorMagnitude) / magnitude);
3719  }
3720  else
3721  {
3722  //Scalar component is small, shouldn't cause loss of precision
3723  angle = ACos(x.w / magnitude);
3724  }
3725 
3726  T newAngle = angle * y;
3727  T div = Sin(newAngle) / Sin(angle);
3728  T mag = Pow(magnitude, y - static_cast<T>(1));
3729  return tQuaternion<T>(Cos(newAngle) * magnitude * mag, x.x * div * mag, x.y * div * mag, x.z * div * mag);
