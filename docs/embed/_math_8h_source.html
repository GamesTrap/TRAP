2496 {

2498  //Needed to prevent a division by 0 error later on
2499  if (y > -Epsilon<T>() && y < Epsilon<T>())
2500  return tQuaternion<T>(1, 0, 0, 0);
2501

2503  T magnitude = Sqrt(x.x * x.x + x.y * x.y + x.z * x.z + x.w * x.w);

2506  if(Abs(x.w / magnitude) > CosOneOverTwo<T>())

2508  //Scalar component is close to 1; using it to recover angle would lose precision
2509  //Instead, we use the non-scalar components since Sin() is accurate around 0
2510

2513  //if(Abs(vectorMagnitude - static_cast<T>(0)) < Epsilon<T>())
2514  //Despite the compiler might say, we actually want to compare vectorMagnitude to 0.
2515  if(vectorMagnitude < std::numeric_limits<T>::min())
2516  {
2517  //Equivalent to raising a real number to a power
2518  return tQuaternion<T>(Pow(x.w, y), 0, 0, 0);

2522  }
2523  else

2526  angle = ACos(x.w / magnitude);
