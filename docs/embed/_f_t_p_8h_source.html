71  ServiceReadySoon = 120, //Service ready in N minutes
72  DataConnectionAlreadyOpened = 125, //Data connection already opened, transfer starting
73  OpeningDataConnection = 150, //File status ok, about to open data connection
74 
75  //2XX: the requested action has been successfully completed
76  OK = 200, //Command OK
77  PointlessCommand = 202, //Command not implemented
78  SystemStatus = 211, //System status, or system help reply
79  DirectoryStatus = 212, //Directory status
80  FileStatus = 213, //File status
81  HelpMessage = 214, //Help message
82  SystemType = 215, //NAME system type, where NAME is an official system name from the list in the Assigned Numbers doc
83  ServiceReady = 220, //Service ready for new user
84  ClosingConnection = 221, //Service closing control connection
85  DataConnectionOpened = 225, //Data connection open, no transfer in progress
86  ClosingDataConnection = 226, //Closing data connection, requested file action successful
87  EnteringPassiveMode = 227, //Entering passive mode
88  LoggedIn = 230, //User logged in, proceed. Logged out if appropriate
89  FileActionOK = 250, //Requested file action OK
90  DirectoryOK = 257, //PATHNAME created
91 
92  //3XX: the command has been accepted, but the requested action
93  // is dormant, pending receipt of further information
94  NeedPassword = 331, //User name OK, need password
95  NeedAccountToLogIn = 332, //Need account for login
96  NeedInformation = 350, //Requested file action pending further information
97 
98  //4XX: the command was not accepted and the requested action did not take place,
99  // but the error condition is temporary and the action may be requested again
100  ServiceUnavailable = 421, //Service not available, closing control connection
101  DataConnectionUnavailable = 425, //Can't open data connection
102  TransferAborted = 426, //Connection closed, transfer aborted
103  FileActionAborted = 450, //Requested file action not taken
104  LocalError = 451, //Requested action aborted, local error in processing
105  InsufficientStorageSpace = 452, //Requested action not taken; insufficient storage space in system, file unavailable
106 
107  //5XX: the command was not accepted and
108  // the requested action did not take place
109  CommandUnknown = 500, //Syntax error, command unrecognized
110  ParametersUnknown = 501, //Syntax error in parameters or arguments
111  CommandNotImplemented = 502, //Command not implemented
112  BadCommandSequence = 503, //Bad sequence of commands
113  ParameterNotImplemented = 504, //Command not implemented for that parameter
114  NotLoggedIn = 530, //Not logged in
115  NeedAccountToStore = 532, //Need account for storing files
116  FileUnavailable = 550, //Requested action not taken, file unavailable
117  PageTypeUnknown = 551, //Requested action aborted, page type unknown
118  NotEnoughMemory = 552, //Requested file action aborted, exceeded storage allocation
119  FilenameNotAllowed = 553, //Requested action not taken, file name not allowed
120 
121  //10XX: custom codes
122  InvalidResponse = 1000, //Not part of the FTP standard, generated by TRAP when a received response cannot be parsed
123  ConnectionFailed = 1001, //Not part of the FTP standard, generated by TRAP when the low-level socket connection with the server fails
124  ConnectionClosed = 1002, //Not part of the FTP standard, generated by TRAP when the low-level socket connection is unexpectedly closed
125  InvalidFile = 1003 //Not part of the FTP standard, generated by TRAP when a local file cannot be read or written
126  };
