84  ServiceReadySoon = 120, //Service ready in N minutes
85  DataConnectionAlreadyOpened = 125, //Data connection already opened, transfer starting
86  OpeningDataConnection = 150, //File status ok, about to open data connection
87 
88  //2XX: the requested action has been successfully completed
89  OK = 200, //Command OK
90  PointlessCommand = 202, //Command not implemented
91  SystemStatus = 211, //System status, or system help reply
92  DirectoryStatus = 212, //Directory status
93  FileStatus = 213, //File status
94  HelpMessage = 214, //Help message
95  SystemType = 215, //NAME system type, where NAME is an official system name from the list in the Assigned Numbers doc
96  ServiceReady = 220, //Service ready for new user
97  ClosingConnection = 221, //Service closing control connection
98  DataConnectionOpened = 225, //Data connection open, no transfer in progress
99  ClosingDataConnection = 226, //Closing data connection, requested file action successful
100  EnteringPassiveMode = 227, //Entering passive mode
101  LoggedIn = 230, //User logged in, proceed. Logged out if appropriate
102  FileActionOK = 250, //Requested file action OK
103  DirectoryOK = 257, //PATHNAME created
104 
105  //3XX: the command has been accepted, but the requested action
106  // is dormant, pending receipt of further information
107  NeedPassword = 331, //User name OK, need password
108  NeedAccountToLogIn = 332, //Need account for login
109  NeedInformation = 350, //Requested file action pending further information
110 
111  //4XX: the command was not accepted and the requested action did not take place,
112  // but the error condition is temporary and the action may be requested again
113  ServiceUnavailable = 421, //Service not available, closing control connection
114  DataConnectionUnavailable = 425, //Can't open data connection
115  TransferAborted = 426, //Connection closed, transfer aborted
116  FileActionAborted = 450, //Requested file action not taken
117  LocalError = 451, //Requested action aborted, local error in processing
118  InsufficientStorageSpace = 452, //Requested action not taken; insufficient storage space in system, file unavailable
119 
120  //5XX: the command was not accepted and
121  // the requested action did not take place
122  CommandUnknown = 500, //Syntax error, command unrecognized
123  ParametersUnknown = 501, //Syntax error in parameters or arguments
124  CommandNotImplemented = 502, //Command not implemented
125  BadCommandSequence = 503, //Bad sequence of commands
126  ParameterNotImplemented = 504, //Command not implemented for that parameter
127  NotLoggedIn = 530, //Not logged in
128  NeedAccountToStore = 532, //Need account for storing files
129  FileUnavailable = 550, //Requested action not taken, file unavailable
130  PageTypeUnknown = 551, //Requested action aborted, page type unknown
131  NotEnoughMemory = 552, //Requested file action aborted, exceeded storage allocation
132  FilenameNotAllowed = 553, //Requested action not taken, file name not allowed
133 
134  //10XX: custom codes
135  InvalidResponse = 1000, //Not part of the FTP standard, generated by TRAP when a received response cannot be parsed
136  ConnectionFailed = 1001, //Not part of the FTP standard, generated by TRAP when the low-level socket connection with the server fails
137  ConnectionClosed = 1002, //Not part of the FTP standard, generated by TRAP when the low-level socket connection is unexpectedly closed
138  InvalidFile = 1003 //Not part of the FTP standard, generated by TRAP when a local file cannot be read or written
139  };
