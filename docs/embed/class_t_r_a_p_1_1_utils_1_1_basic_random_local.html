template<typename engine, typename Seeder = SeederDefault, template< typename > class IntegerDist = std::uniform_int_distribution, template< typename > class RealDist = std::uniform_real_distribution, typename BoolDist = std::bernoulli_distribution>
class TRAP::Utils::BasicRandomLocal< engine, Seeder, IntegerDist, RealDist, BoolDist >

Base template class for random with local API and local internal member storage.
It IS thread safe but less efficient then BasicRandomStatic.

Template Parameters
engineRandom engine with interface like in the std::mt19937.
SeederSeeder type which return seed for internal engine through operator().
