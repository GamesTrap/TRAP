158  template<std::size_t N, std::size_t K>
159  constexpr auto CleanupOutputString(const char(&expr)[N], const char(&remove)[K])
160  {
161  ChangeResult<N> result = {};
162 
163  std::size_t srcIndex = 0;
164  std::size_t dstIndex = 0;
165  while(srcIndex < N)
166  {
167  std::size_t matchIndex = 0;
168  while (matchIndex < K - 1 && srcIndex + matchIndex < N - 1 && expr[srcIndex + matchIndex] == remove[matchIndex])
169  matchIndex++;
170  if (matchIndex == K - 1)
171  srcIndex += matchIndex;
172  result.Data[dstIndex++] = expr[srcIndex] == '"' ? '\'' : expr[srcIndex];
173  srcIndex++;
174  }
175 
176  return result;
177  }
