169  {
170  //2XX: Success
171  OK = 200, //Most common code returned when operation was successful
172  Created = 201, //The resource has successfully been created
173  Accepted = 202, //The request has been accepted, but will be processed later by the server
174  NoContent = 204, //The server didn't send any data in return
175  ResetContent = 205, //The server informs the client that it should clear the view (form) that caused the request to be sent
176  PartialContent = 206, //The server has sent a part of the resource, as a response to a partial GET request
177 
178  //3XX: Redirection
179  MultipleChoices = 300, //The requested page can be accessed from several locations
180  MovedPermanently = 301, //The requested page has permanently moved to a new location
181  MovedTemporarily = 302, //The requested page has temporarily moved to a new location
182  NotModified = 304, //For conditional requests, means the requested page hasn't changed and doesn't need to be refreshed
183 
184  //4XX: Client error
185  BadRequest = 400, //The server couldn't understand the request (syntax error)
186  Unauthorized = 401, //The requested page needs an authentication to be accessed
187  Forbidden = 403, //The requested page cannot be accessed at all, event with authentication
188  NotFound = 404, //The requested page doesn't exist
189  RangeNotSatisfiable = 407, //The server can't satisfy the partial GET request (with a "Range" header field)
190 
191  //5XX: Server error
192  InternalServerError = 500, //The server encountered an unexpected error
193  NotImplemented = 501, //The server doesn't implement a requested feature
194  BadGateway = 502, //The gateway server has received an error from the source server
195  ServiceNotAvailable = 503, //The server is temporarily unavailable (overloaded, in maintenance, ...)
196  GatewayTimeout = 504, //The gateway server couldn't receive a response from the source server
197  VersionNotSupported = 505, //The server doesn't support the requested HTTP version
198 
199  //10XX: Custom codes
200  InvalidResponse = 1000, //Response is not a valid HTTP one
201  ConnectionFailed = 1001 //Connection with server failed
202  };
