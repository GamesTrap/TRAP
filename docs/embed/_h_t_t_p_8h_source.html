183  class Response
184  {
185  public:
189  enum class Status
190  {
191  //2XX: success
192  OK = 200, //MOst common code returned when operation was successful
193  Created = 201, //The resource has successfully been created
194  Accepted = 202, //The request has been accepted, but will be processed later by the server
195  NoContent = 204, //The server didn't send any data in return
196  ResetContent = 205, //The server informs the client that it should clear the view (form) that caused the request to be sent
197  PartialContent = 206, //The server has sent a part of the resource, as a response to a partial GET request
198 
199  //3XX: redirection
200  MultipleChoices = 300, //The requested page can be accessed from several locations
201  MovedPermanently = 301, //The requested page has permanently moved to a new location
202  MovedTemporarily = 302, //The requested page has temporarily moved to a new location
203  NotModified = 304, //For conditional requests, means the requested page hasn't changed and doesn't need to be refreshed
204 
205  //4XX: client error
206  BadRequest = 400, //The server couldn't understand the request (syntax error)
207  Unauthorized = 401, //The requested page needs an authentication to be accessed
208  Forbidden = 403, //The requested page cannot be accessed at all, event with authentication
209  NotFound = 404, //The requested page doesn't exist
210  RangeNotSatisfiable = 407, //The server can't satisfy the partial GET request (with a "Range" header field)
211 
212  //5XX: server error
213  InternalServerError = 500, //The server encountered an unexpected error
214  NotImplemented = 501, //The server doesn't implement a requested feature
215  BadGateway = 502, //The gateway server has received an error from the source server
216  ServiceNotAvailable = 503, //The server is temporarily unavailable (overloaded, in maintenance, ...)
217  GatewayTimeout = 504, //The gateway server couldn't receive a response from the source server
218  VersionNotSupported = 505, //The server doesn't support the requested HTTP version
219 
220  //10XX: custom codes
221  InvalidResponse = 1000, //Response is not a valid HTTP one
222  ConnectionFailed = 1001 //Connection with server failed
223  };
