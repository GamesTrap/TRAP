248 template <typename T>
249 TRAP::Math::tQuat<T>::tQuat(const Vec<3, T>& u, const Vec<3, T>& v)
250 {
251  //TODO Can't use TRAP::Math::Sqrt here
252  T normUNormV = std::sqrt(Dot(u, u) * Dot(v, v));
253  T realPart = normUNormV + Dot(u, v);
254  Vec<3, T> t;
255 
256  if(realPart < static_cast<T>(1.e-6f) * normUNormV)
257  {
258  //If u and v are exactly opposite, rotate 180 degrees around an arbitrary orthogonal axis.
259  //Axis normalization can happen later, when we normalize the quaternion.
260  realPart = static_cast<T>(0);
261  //TODO Can't use TRAP::Math::Abs here
262  t = std::abs(u.x) > std::abs(u.z) ? Vec<3, T>(-u.y, u.x, static_cast<T>(0)) : Vec<3, T>(static_cast<T>(0), -u.z, u.y);
263  }
264  else
265  {
266  //Otherwise, build quaternion the standard way.
267  t = Cross(u, v);
268  }
269 
270  *this = Normalize(tQuat<T>(realPart, t.x, t.y, t.z));
271 }

276 template <typename T>
277 constexpr TRAP::Math::tQuat<T>::tQuat(const Vec<3, T>& eulerAnglesInRadians)
278 {
279  Vec<3, T> c = Cos(eulerAnglesInRadians * T(0.5));
280  Vec<3, T> s = Sin(eulerAnglesInRadians * T(0.5));
281 
282  this->w = c.x * c.y * c.z + s.x * s.y * s.z;
283  this->x = s.x * c.y * c.z - c.x * s.y * s.z;
284  this->y = c.x * s.y * c.z + s.x * c.y * s.z;
285  this->z = c.x * c.y * s.z - s.x * s.y * c.z;
286 }

333 template <typename T>
334 template <typename U>
335 constexpr TRAP::Math::tQuat<T>& TRAP::Math::tQuat<T>::operator*=(const tQuat<U>& r)
336 {
337  const tQuat<T> p(*this);
338  const tQuat<T> q(r);
339 
340  this->w = p.w * q.w - p.x * q.x - p.y * q.y - p.z * q.z;
341  this->x = p.w * q.x + p.x * q.w + p.y * q.z - p.z * q.y;
342  this->y = p.w * q.y + p.y * q.w + p.z * q.x - p.x * q.z;
343  this->z = p.w * q.z + p.z * q.w + p.x * q.y - p.y * q.x;
344 
345  return *this;
346 }

448 template <typename T>
449 constexpr TRAP::Math::Vec<3, T> TRAP::Math::operator*(const tQuat<T>& q, const Vec<3, T>& v)
450 {
451  const Vec<3, T> quaternionVector(q.x, q.y, q.z);
452  const Vec<3, T> uv(Cross(quaternionVector, v));
453  const Vec<3, T> uuv(Cross(quaternionVector, uv));
454 
455  return v + ((uv * q.w) + uuv) * static_cast<T>(2);
456 }
