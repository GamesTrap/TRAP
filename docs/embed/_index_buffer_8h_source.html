143 template<typename T>
144 TRAP::Scope<TRAP::Graphics::IndexBuffer> TRAP::Graphics::IndexBuffer::Init(T* indices, const uint64_t size,
145  const UpdateFrequency updateFrequency)
146 {
147  static_assert(std::is_same_v<T, uint16_t> || std::is_same_v<T, uint32_t>,
148  "Trying to initialize index buffer with wrong index type!");
149 
150  RendererAPI::IndexType indexType = {};
151  if(indices)
152  {
153  if constexpr(std::is_same_v<T, uint16_t>)
154  indexType = RendererAPI::IndexType::UInt16;
155  else if constexpr(std::is_same_v<T, uint32_t>)
156  indexType = RendererAPI::IndexType::UInt32;
157  }
158  else
159  {
160  if(size / sizeof(uint16_t) < std::numeric_limits<uint16_t>::max())
161  indexType = RendererAPI::IndexType::UInt16;
162  else
163  indexType = RendererAPI::IndexType::UInt32;
164  }
165 
166  TRAP::Scope<IndexBuffer> buffer = TRAP::Scope<IndexBuffer>(new IndexBuffer(indexType));
167 
168  RendererAPI::BufferLoadDesc desc{};
169  desc.Desc.MemoryUsage = (updateFrequency == UpdateFrequency::Static) ? RendererAPI::ResourceMemoryUsage::GPUOnly :
170  RendererAPI::ResourceMemoryUsage::CPUToGPU;
171  desc.Desc.Descriptors = RendererAPI::DescriptorType::IndexBuffer;
172  desc.Desc.Size = size;
173  desc.Desc.Flags = (updateFrequency != UpdateFrequency::Static) ? RendererAPI::BufferCreationFlags::PersistentMap :
174  RendererAPI::BufferCreationFlags::None;
175  desc.Data = indices;
176 
177  RendererAPI::GetResourceLoader()->AddResource(desc, &buffer->m_token);
178 
179  buffer->m_indexBuffer = desc.Buffer;
180 
181  return buffer;
182 }

187 void TRAP::Graphics::IndexBuffer::SetDataInternal(const T* indices, const uint64_t size,
188  const uint64_t offset)
189 {
190  static_assert(std::is_same_v<T, uint16_t> || std::is_same_v<T, uint32_t>,
191  "Trying to initialize index buffer with wrong index type!");
192 
193  TRAP_ASSERT(indices);
194  TRAP_ASSERT(size + offset <= m_indexBuffer->GetSize());
195 
196  RendererAPI::BufferUpdateDesc desc{};
197  desc.Buffer = m_indexBuffer;
198  desc.DstOffset = offset;
199  RendererAPI::GetResourceLoader()->BeginUpdateResource(desc);
200  std::memcpy(desc.MappedData, indices, size);
201  RendererAPI::GetResourceLoader()->EndUpdateResource(desc, &m_token);
202 
203  if constexpr(std::is_same_v<T, uint16_t>)
204  m_indexType = RendererAPI::IndexType::UInt16;
205  else if constexpr(std::is_same_v<T, uint32_t>)
206  m_indexType = RendererAPI::IndexType::UInt32;
207 }
