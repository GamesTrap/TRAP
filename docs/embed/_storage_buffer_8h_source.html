131 inline void TRAP::Graphics::StorageBuffer::GetData(const T* data, const uint64_t size, const uint64_t offset, Window* window)
132 {
133  TRAP_ASSERT(size + offset <= m_storageBuffers[0]->GetSize());
134 
135  if(!window)
136  window = TRAP::Application::GetWindow();
137 
138  RendererAPI::BufferUpdateDesc desc{};
139  const uint32_t imageIndex = GetUpdateFrequency() ==
140  RendererAPI::DescriptorUpdateFrequency::Static ?
141  0 : RendererAPI::GetCurrentImageIndex(window);
142  desc.Buffer = m_storageBuffers[imageIndex];
143  desc.DstOffset = offset;
144  RendererAPI::GetResourceLoader()->BeginUpdateResource(desc);
145  std::memcpy(data, desc.MappedData, size);
146  RendererAPI::GetResourceLoader()->EndUpdateResource(desc, &m_tokens[imageIndex]);
147 }
