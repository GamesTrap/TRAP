219 inline TRAP::Scope<TRAP::Image> TRAP::Image::FlipX(const Scope<Image>& img)
220 {
221  if (img)
222  {
223  Scope<Image> result;
224 
225  if (img->IsHDR())
226  {
227  const std::vector<float> flipped = FlipX(img->GetWidth(), img->GetHeight(), img->GetColorFormat(), static_cast<const float*>(img->GetPixelData()));
228 
229  result = LoadFromMemory(img->GetWidth(), img->GetHeight(), img->GetColorFormat(), flipped);
230  }
231  else if ((img->IsImageGrayScale() && img->GetBitsPerPixel() == 16 && !img->HasAlphaChannel()) ||
232  (img->IsImageGrayScale() && img->GetBitsPerPixel() == 32 && img->HasAlphaChannel()) ||
233  (img->IsImageColored() && img->GetBitsPerPixel() == 48 && !img->HasAlphaChannel()) ||
234  (img->IsImageColored() && img->GetBitsPerPixel() == 64 && img->HasAlphaChannel()))
235  {
236  const std::vector<uint16_t> flipped = FlipX(img->GetWidth(), img->GetHeight(), img->GetColorFormat(), static_cast<const uint16_t*>(img->GetPixelData()));
237 
238  result = LoadFromMemory(img->GetWidth(), img->GetHeight(), img->GetColorFormat(), flipped);
239  }
240  else
241  {
242  const std::vector<uint8_t> flipped = FlipX(img->GetWidth(), img->GetHeight(), img->GetColorFormat(), static_cast<const uint8_t*>(img->GetPixelData()));
243 
244  result = LoadFromMemory(img->GetWidth(), img->GetHeight(), img->GetColorFormat(), flipped);
245  }
246 
247  return result;
248  }
249 
250  return nullptr;

291 template <typename T>
292 std::vector<T> TRAP::Image::FlipX(const uint32_t width, const uint32_t height, const ColorFormat format, const T* data)
293 {
294  if (!(std::is_same<T, uint8_t>::value || std::is_same<T, uint16_t>::value || std::is_same<T, float>::value))
295  {
296  TRAP_ASSERT(false, "Invalid type!");
297  return {};
298  }
299  if (format == ColorFormat::NONE || !data)
300  return {};
301 
302  std::vector<T> newData{};
303  uint32_t stride = 0;
304  if (format == ColorFormat::GrayScale)
305  {
306  newData.assign(data, data + static_cast<uint64_t>(width) * static_cast<uint64_t>(height));
307  stride = height;
308  }
309  else if (format == ColorFormat::GrayScaleAlpha)
310  {
311  newData.assign(data, data + static_cast<uint64_t>(width) * static_cast<uint64_t>(height) * 2);
312  stride = height * 2;
313  }
314  else if (format == ColorFormat::RGB)
315  {
316  newData.assign(data, data + static_cast<uint64_t>(width) * static_cast<uint64_t>(height) * 3);
317  stride = height * 3;
318  }
319  else if (format == ColorFormat::RGBA)
320  {
321  newData.assign(data, data + static_cast<uint64_t>(width) * static_cast<uint64_t>(height) * 4);
322  stride = height * 4;
323  }
324 
325  std::vector<T> row{};
326  row.resize(stride);
327  uint32_t lowOffset = 0;
328  uint32_t highOffset = (width - 1) * stride;
329 
330  for (; lowOffset < highOffset; lowOffset += stride, highOffset -= stride)
331  {
332  std::copy(newData.data() + lowOffset, newData.data() + lowOffset + stride, row.data());
333  std::copy(newData.data() + highOffset, newData.data() + highOffset + stride, newData.data() + lowOffset);
334  std::copy(row.data(), row.data() + stride, newData.data() + highOffset);
335  }
336 
337  return newData;
338 }
