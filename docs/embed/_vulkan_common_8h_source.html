364 constexpr bool TRAP::Graphics::API::ErrorCheck(VkResult result, const char* function, const char* file, int32_t line)
365 {
366  if(result >= 0)
367  return true;
368 
369  switch (result)
370  {
371  case VK_ERROR_OUT_OF_HOST_MEMORY:
372  TP_ERROR(Log::RendererVulkanPrefix, "VK_ERROR_OUT_OF_HOST_MEMORY: ", function, " @[", file, ':', line, ']');
373  break;
374  case VK_ERROR_OUT_OF_DEVICE_MEMORY:
375  TP_ERROR(Log::RendererVulkanPrefix, "VK_ERROR_OUT_OF_DEVICE_MEMORY: ", function, " @[", file, ':', line,
376  ']');
377  break;
378  case VK_ERROR_INITIALIZATION_FAILED:
379  TP_ERROR(Log::RendererVulkanPrefix, "VK_ERROR_INITIALIZATION_FAILED: ", function, " @[", file, ':', line,
380  ']');
381  break;
382  case VK_ERROR_DEVICE_LOST:
383  TP_ERROR(Log::RendererVulkanPrefix, "VK_ERROR_DEVICE_LOST: ", function, " @[", file, ':', line, ']');
384  break;
385  case VK_ERROR_MEMORY_MAP_FAILED:
386  TP_ERROR(Log::RendererVulkanPrefix, "VK_ERROR_MEMORY_MAP_FAILED: ", function, " @[", file, ':', line, ']');
387  break;
388  case VK_ERROR_LAYER_NOT_PRESENT:
389  TP_ERROR(Log::RendererVulkanPrefix, "VK_ERROR_LAYER_NOT_PRESENT: ", function, " @[", file, ':', line, ']');
390  break;
391  case VK_ERROR_EXTENSION_NOT_PRESENT:
392  TP_ERROR(Log::RendererVulkanPrefix, "VK_ERROR_EXTENSION_NOT_PRESENT: ", function, " @[", file, ':', line,
393  ']');
394  break;
395  case VK_ERROR_FEATURE_NOT_PRESENT:
396  TP_ERROR(Log::RendererVulkanPrefix, "VK_ERROR_FEATURE_NOT_PRESENT: ", function, " @[", file, ':', line, ']');
397  break;
398  case VK_ERROR_INCOMPATIBLE_DRIVER:
399  TP_ERROR(Log::RendererVulkanPrefix, "VK_ERROR_INCOMPATIBLE_DRIVER: ", function, " @[", file, ':', line, ']');
400  break;
401  case VK_ERROR_TOO_MANY_OBJECTS:
402  TP_ERROR(Log::RendererVulkanPrefix, "VK_ERROR_TOO_MANY_OBJECTS: ", function, " @[", file, ':', line, ']');
403  break;
404  case VK_ERROR_FORMAT_NOT_SUPPORTED:
405  TP_ERROR(Log::RendererVulkanPrefix, "VK_ERROR_FORMAT_NOT_SUPPORTED: ", function, " @[", file, ':', line,
406  ']');
407  break;
408  case VK_ERROR_SURFACE_LOST_KHR:
409  TP_ERROR(Log::RendererVulkanPrefix, "VK_ERROR_SURFACE_LOST_KHR: ", function, " @[", file, ':', line, ']');
410  break;
411  case VK_ERROR_NATIVE_WINDOW_IN_USE_KHR:
412  TP_ERROR(Log::RendererVulkanPrefix, "VK_ERROR_NATIVE_WINDOW_IN_USE_KHR: ", function, " @[", file, ':', line,
413  ']');
414  break;
415  case VK_SUBOPTIMAL_KHR:
416  TP_ERROR(Log::RendererVulkanPrefix, "VK_SUBOPTIMAL_KHR: ", function, " @[", file, ':', line, ']');
417  break;
418  case VK_ERROR_OUT_OF_DATE_KHR:
419  TP_ERROR(Log::RendererVulkanPrefix, "VK_ERROR_OUT_OF_DATE_KHR: ", function, " @[", file, ':', line, ']');
420  break;
421  case VK_ERROR_INCOMPATIBLE_DISPLAY_KHR:
422  TP_ERROR(Log::RendererVulkanPrefix, "VK_ERROR_INCOMPATIBLE_DISPLAY_KHR: ", function, " @[", file, ':', line,
423  ']');
424  break;
425  case VK_ERROR_VALIDATION_FAILED_EXT:
426  TP_ERROR(Log::RendererVulkanPrefix, "VK_ERROR_VALIDATION_FAILED_EXT: ", function, " @[", file, ':', line,
427  ']');
428  break;
429 
430  default:
431  TP_ERROR(Log::RendererVulkanPrefix, "Unknown error: ", function, " @[", file, ':', line, ']');
432  break;
433  }
434 
435  return false;
436 }
