374 std::array<TRAP::Scope<TRAP::Image>, 6> TRAP::Graphics::Texture::SplitImageFromCross(const TRAP::Image* const image)
375 {
376  const bool isHorizontal = image->GetWidth() > image->GetHeight();
377 
378  const uint32_t stride = image->GetBytesPerPixel();
379  uint32_t face = 0;
380  uint32_t cxLimit = 4, cyLimit = 3;
381  if(!isHorizontal)
382  {
383  cxLimit = 3;
384  cyLimit = 4;
385  }
386  uint32_t faceWidth = image->GetWidth() / cxLimit;
387  uint32_t faceHeight = image->GetHeight() / cyLimit;
388 
389  std::array<std::vector<T>, 6> cubeTextureData;
390  for(auto& i : cubeTextureData)
391  i.resize(faceWidth * faceHeight * stride);
392 
393  for(uint32_t cy = 0; cy < cyLimit; ++cy)
394  {
395  for(uint32_t cx = 0; cx < cxLimit; ++cx)
396  {
397  if((cy == 0 || cy == 2 || cy == 3) && cx != 1)
398  continue;
399 
400  for(uint32_t y = 0; y < faceHeight; ++y)
401  {
402  uint32_t offset = y;
403  if(!isHorizontal && face == 5)
404  offset = faceHeight - (y + 1);
405  const uint32_t yp = cy * faceHeight + offset;
406 
407  for(uint32_t x = 0; x < faceWidth; ++x)
408  {
409  offset = x;
410  if(!isHorizontal && face == 5)
411  offset = faceWidth - (x + 1);
412  const uint32_t xp = cx * faceWidth + offset;
413  switch(stride)
414  {
415  case 1:
416  cubeTextureData[face][(x + y * faceWidth) * stride + 0] = static_cast<const T*>
417  (
418  image->GetPixelData()
419  )[(xp + yp * image->GetWidth()) * stride + 0];
420  break;
421 
422  case 2:
423  cubeTextureData[face][(x + y * faceWidth) * stride + 0] = static_cast<const T*>
424  (
425  image->GetPixelData()
426  )[(xp + yp * image->GetWidth()) * stride + 0];
427  cubeTextureData[face][(x + y * faceWidth) * stride + 1] = static_cast<const T*>
428  (
429  image->GetPixelData()
430  )[(xp + yp * image->GetWidth()) * stride + 1];
431  break;
432 
433  case 3:
434  cubeTextureData[face][(x + y * faceWidth) * stride + 0] = static_cast<const T*>
435  (
436  image->GetPixelData()
437  )[(xp + yp * image->GetWidth()) * stride + 0];
438  cubeTextureData[face][(x + y * faceWidth) * stride + 1] = static_cast<const T*>
439  (
440  image->GetPixelData()
441  )[(xp + yp * image->GetWidth()) * stride + 1];
442  cubeTextureData[face][(x + y * faceWidth) * stride + 2] = static_cast<const T*>
443  (
444  image->GetPixelData()
445  )[(xp + yp * image->GetWidth()) * stride + 2];
446  break;
447 
448  case 4:
449  cubeTextureData[face][(x + y * faceWidth) * stride + 0] = static_cast<const T*>
450  (
451  image->GetPixelData()
452  )[(xp + yp * image->GetWidth()) * stride + 0];
453  cubeTextureData[face][(x + y * faceWidth) * stride + 1] = static_cast<const T*>
454  (
455  image->GetPixelData()
456  )[(xp + yp * image->GetWidth()) * stride + 1];
457  cubeTextureData[face][(x + y * faceWidth) * stride + 2] = static_cast<const T*>
458  (
459  image->GetPixelData()
460  )[(xp + yp * image->GetWidth()) * stride + 2];
461  cubeTextureData[face][(x + y * faceWidth) * stride + 3] = static_cast<const T*>
462  (
463  image->GetPixelData()
464  )[(xp + yp * image->GetWidth()) * stride + 3];
465  break;
466 
467  default:
468  break;
469  }
470  }
471  }
472  ++face;
473  }
474  }
475 
476  std::array<TRAP::Scope<TRAP::Image>, 6> images{};
477 
478  //Load Images in correct order
479  if(cxLimit == 4 && cyLimit == 3)
480  {
481  images[0] = TRAP::Image::LoadFromMemory(faceWidth, faceHeight, image->GetColorFormat(), cubeTextureData[3]); //+X
482  images[1] = TRAP::Image::LoadFromMemory(faceWidth, faceHeight, image->GetColorFormat(), cubeTextureData[1]); //-X
483  images[2] = TRAP::Image::LoadFromMemory(faceWidth, faceHeight, image->GetColorFormat(), cubeTextureData[0]); //+Y
484  images[3] = TRAP::Image::LoadFromMemory(faceWidth, faceHeight, image->GetColorFormat(), cubeTextureData[5]); //-Y
485  images[4] = TRAP::Image::LoadFromMemory(faceWidth, faceHeight, image->GetColorFormat(), cubeTextureData[2]); //+Z
486  images[5] = TRAP::Image::LoadFromMemory(faceWidth, faceHeight, image->GetColorFormat(), cubeTextureData[4]); //-Z
487  }
488  else
489  {
490  images[0] = TRAP::Image::LoadFromMemory(faceWidth, faceHeight, image->GetColorFormat(), cubeTextureData[2]); //+X
491  images[1] = TRAP::Image::LoadFromMemory(faceWidth, faceHeight, image->GetColorFormat(), cubeTextureData[5]); //-X
492  images[2] = Rotate90CounterClockwise(TRAP::Image::LoadFromMemory(faceWidth, faceHeight, image->GetColorFormat(), cubeTextureData[0]).get()); //+Y
493  images[3] = Rotate90Clockwise(TRAP::Image::LoadFromMemory(faceWidth, faceHeight, image->GetColorFormat(), cubeTextureData[4]).get()); //-Y
494  images[4] = TRAP::Image::LoadFromMemory(faceWidth, faceHeight, image->GetColorFormat(), cubeTextureData[1]); //+Z
495  images[5] = TRAP::Image::LoadFromMemory(faceWidth, faceHeight, image->GetColorFormat(), cubeTextureData[3]); //-Z
496  }
497 
498  return images;
499 }
